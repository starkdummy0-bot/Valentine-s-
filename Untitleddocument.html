<script>

let current = 0;

const slides = document.querySelectorAll(".slide");

function nextSlide() {

  slides[current].classList.remove("active");

  current++;

  slides[current].classList.add("active");

}

function slide1No() {

  document.getElementById("r1").innerText =

    "Nooo 🥺 please be free, you’re making me dramatic!";

}

function slide1Yes() {

  confetti();

  setTimeout(nextSlide, 800);

}

function slide2No() {

  document.getElementById("r2").innerText =

    "Come on 😔 I promise it’s cute.";

}

function slide2Yes() {

  confetti();

  setTimeout(nextSlide, 800);

}

function slide3No() {

  document.getElementById("r3").innerText =

    "WOW 😭 okay that hurt… I’m being very dramatic.";

}

function slide3Yes() {

  document.getElementById("r3").innerText =

    "HA! I knew it 😏";

  setTimeout(nextSlide, 900);

}

function slide4No() {

  document.getElementById("r4").innerText =

    "Okay wow… I’m being extra dramatic right now 😭";

}

function slide4Yes() {

  confetti();

  document.getElementById("r4").innerText =

    "I love you the most 🥰";

  setTimeout(nextSlide, 1200);

}

function countdown() {

  let i = 3;

  const el = document.getElementById("count");

  el.innerText = i;

  const timer = setInterval(() => {

    i--;

    if (i === 0) {

      clearInterval(timer);

      el.innerText = "✨";

      nextSlide();

    } else {

      el.innerText = i;

    }

  }, 700);

}

function finalYes() {

  confetti();

  setTimeout(nextSlide, 1000);

}

function confetti() {

  for (let i = 0; i < 40; i++) {

    const c = document.createElement("div");

    c.className = "confetti";

    c.style.left = Math.random() * 100 + "vw";

    c.style.background =

      ["#ff9ecf", "#ffd6e8", "#cfa6ff"][Math.floor(Math.random()*3)];

    document.body.appendChild(c);

    setTimeout(() => c.remove(), 3000);

  }

}

// floating emojis

setInterval(() => {

  const e = document.createElement("div");

  e.className = "emoji";

  e.innerText = ["💖","✨","🌸","🫶"][Math.floor(Math.random()*4)];

  e.style.left = Math.random() * 100 + "vw";

  document.body.appendChild(e);

  setTimeout(() => e.remove(), 6000);

}, 900);
